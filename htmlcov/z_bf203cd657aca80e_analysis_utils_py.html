274def convert_odds_to_implied_probabilities( 

275 odds_list: list[float], 

276) -> tuple[list[float], float]: 

277 """ 

278 Converts odds to implied probabilities (no vigorish) and calculates overround. 

279 

280 Args: 

281 odds_list: A list of decimal odds (e.g., [2.5, 3.0, 4.0]). 

282 

283 Returns: 

284 A tuple containing: 

285 - A list of implied probabilities corresponding to the input odds. 

286 - The calculated overround for the market. 

287 """ 

288 if not odds_list: 

289 return [], 0.0 

290 

291 raw_probabilities = [] 

292 for odds in odds_list: 

293 if odds > 1: # Odds must be greater than 1 

294 raw_probabilities.append(1 / odds) 

295 else: 

296 raw_probabilities.append(0.0) # Treat invalid odds as 0 probability 

297 

298 overround = sum(raw_probabilities) 

299 

300 if overround == 0: 

301 # Avoid division by zero, return uniform probabilities if all odds are invalid 

302 implied_probabilities = [1.0 / len(odds_list)] * len(odds_list) 

303 else: 

304 implied_probabilities = [p / overround for p in raw_probabilities] 

305 

306 return implied_probabilities, overround
